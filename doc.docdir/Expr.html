Module Expr

module Expr: sig .. end
Module with expression type and to_string function implementation
type expr = [ `And of expr * expr
       | `End
       | `Eq of expr * expr
       | `Error of string
       | `False
       | `Frame of expr * expr list
       | `Goal of expr * expr
       | `Impl of expr * expr
       | `Name of string
       | `Neg of expr
       | `Or of expr * expr
       | `Paren of expr
       | `Prog of expr * expr
       | `Proof of expr list
       | `True
       | `Var of char ] 
val to_string : ([< `And of 'a * 'a
         | `End
         | `Eq of 'a * 'a
         | `Error of string
         | `False
         | `Frame of 'a * 'a list
         | `Goal of 'a * 'a
         | `Impl of 'a * 'a
         | `Name of string
         | `Neg of 'a
         | `Or of 'a * 'a
         | `Paren of 'a
         | `Prog of 'a * 'a
         | `Proof of 'a list
         | `True
         | `Var of char
         > `And `Impl ]
        as 'a) ->
       string
Function to_string converts each expression to string
